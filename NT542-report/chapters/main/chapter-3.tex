\section{Phase 1 – Core Security Foundations}

\subsection{System Information Gathering}
\vspace{5pt}
\noindent \textbf{Mô tả:} Thu thập thông tin chi tiết về hệ thống để đánh giá trạng thái bảo mật hiện tại và xác định các thành phần cần cấu hình. Quá trình này bao gồm việc kiểm tra phiên bản hệ điều hành, thông tin phần cứng, các dịch vụ đang chạy và cấu hình mạng.

\vspace{5pt}
\noindent \textbf{Nội dung triển khai:}
\begin{itemize}
\item Kiểm tra phiên bản hệ điều hành Ubuntu và kernel
\item Thu thập thông tin phần cứng (CPU, RAM, disk)
\item Liệt kê các dịch vụ đang chạy và cổng mở
\item Kiểm tra cấu hình mạng và interface
\item Xác định người dùng và nhóm hệ thống
\item Kiểm tra các gói phần mềm đã cài đặt
\end{itemize}

\section{Phase 2 – Filesystem \& System Hardening}
\phaserenumber{2}
\subsection{Filesystem Security}
\vspace{5pt}
\noindent \textbf{Mô tả:} Thực hiện hardening filesystem để bảo vệ các tệp tin và thư mục quan trọng, ngăn chặn truy cập trái phép và giảm thiểu các lỗ hổng bảo mật liên quan đến quyền truy cập file.

\vspace{5pt}
\noindent \textbf{Nội dung triển khai:}
\begin{itemize}
\item Vô hiệu hóa các filesystem không cần thiết (cramfs, squashfs, udf)
\item Cấu hình quyền truy cập cho các file hệ thống quan trọng (/etc/passwd, /etc/shadow, /etc/group, /etc/gshadow)
\item Bảo mật các file cron và thư mục cron jobs
\item Triển khai sticky bit trên các thư mục world-writable
\item Audit và phân tích các SUID/SGID files
\item Kiểm tra các world-writable files và unowned files
\end{itemize}

\vspace{5pt}
\noindent \textbf{Cấu hình sử dụng:}

\begin{table}[H]
\centering
\begin{tabular}{|p{\linewidth}|}
\hline
\textbf{1. Disable Unnecessary Filesystems:} \\
\texttt{modprobe -r cramfs} \# Unload cramfs module \\
\texttt{modprobe -r squashfs} \# Unload squashfs module \\
\texttt{modprobe -r udf} \# Unload udf module \\
\texttt{echo "install cramfs /bin/true" >> /etc/modprobe.d/disable-filesystems.conf} \\
\hline
\end{tabular}
\caption{Lệnh vô hiệu hóa filesystem không cần thiết}
\label{tab:filesystem_disable_cmd}
\end{table}

\begin{table}[H]
\centering
\begin{tabular}{|p{\linewidth}|}
\hline
\textbf{2. Secure Critical Files:} \\
\texttt{chmod 644 /etc/passwd /etc/group} \# Secure user/group files \\
\texttt{chmod 600 /etc/shadow /etc/gshadow} \# Secure password files \\
\texttt{chmod 644 /etc/hosts} \# Secure hosts file \\
\texttt{chown root:root /etc/passwd /etc/shadow /etc/group /etc/gshadow} \\
\hline
\end{tabular}
\caption{Lệnh bảo mật file hệ thống quan trọng}
\label{tab:filesystem_secure_cmd}

\begin{tabular}{|p{\linewidth}|}

\texttt{chmod 600 /etc/crontab} \# Secure main crontab \\
\texttt{chmod 700 /var/spool/cron/atjobs} \# Secure at jobs directory \\
\texttt{chown root:root /etc/crontab} \# Set ownership \\
\hline
\end{tabular}

\subsection{SUID/SGID Audit Report Creation}
\vspace{5pt}
\noindent \textbf{Mô tả:} Thực hiện audit chi tiết các file có quyền SUID (Set User ID) và SGID (Set Group ID) để xác định các file thực thi có quyền nâng cao, có thể tạo ra rủi ro bảo mật nếu bị khai thác.

\vspace{5pt}
\noindent \textbf{Nội dung triển khai:}
\begin{itemize}
\item Sử dụng lệnh find để tìm kiếm SUID files (permission 4000)
\item Sử dụng lệnh find để tìm kiếm SGID files (permission 2000)
\item Phân tích và đánh giá các file được phát hiện
\item Tạo báo cáo audit chi tiết với danh sách đầy đủ

\subsection{System Settings}
\vspace{5pt}
\noindent \textbf{Mô tả:} Cấu hình các tham số bảo mật kernel để tăng cường bảo mật hệ thống ở cấp độ thấp, ngăn chặn các tấn công side-channel và giảm thiểu thông tin lộ ra ngoài.

\vspace{5pt}
\noindent \textbf{Nội dung triển khai:}
\begin{itemize}
\item Cấu hình kernel.dmesg\_restrict để hạn chế truy cập kernel messages
\item Cấu hình kernel.kptr\_restrict để bảo vệ kernel pointers
\item Áp dụng các thiết lập bảo mật kernel thông qua sysctl
\item Kiểm tra và xác nhận các thiết lập đã được áp dụng
\end{itemize}

\texttt{sysctl -p /etc/sysctl.d/99-security.conf} \# Áp dụng config \\
\texttt{sysctl kernel.dmesg\_restrict} \# Kiểm tra thiết lập \\
\texttt{sysctl kernel.kptr\_restrict} \# Kiểm tra thiết lập \\

\end{itemize}

\subsection{ASLR Configuration}

\vspace{5pt}
\noindent \textbf{Nội dung triển khai:}
\begin{itemize}
\item Kích hoạt ASLR với chế độ full randomization
\item Cấu hình kernel.randomize\_va\_space tham số
\item Áp dụng cấu hình ASLR thông qua sysctl

\item \textbf{Đánh giá:} ASLR full randomization cung cấp bảo vệ mạnh mẽ chống lại memory corruption attacks. Với value=2, stack, mmap, VDSO và heap đều được ngẫu nhiên hóa, tăng đáng kể độ khó cho attacker trong việc thực hiện các tấn công exploit.
\end{itemize}

\section{Phase 4 – SSH Hardening}

\begin{itemize}

\item Cấu hình password authentication và allowed users để giới hạn quyền truy cập
\item Thiết lập MaxAuthTries để giới hạn số lần thử đăng nhập
\item Cấu hình ClientAliveInterval để tự động ngắt kết nối không hoạt động
\item Khởi động lại SSH service và tạo báo cáo validation
